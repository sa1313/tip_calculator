<script>
function calculate() {
  var bill_num = document.getElementById("bill_amount").value;
  var tip_num = document.getElementById("tip_percent").value;
  var tip_total = (tip_num/100)*bill_num;
  var total_num = bill_num*(1+(tip_num/100));
  
  
  document.getElementById("tip_total").innerHTML = tip_total;
  document.getElementById("total").innerHTML = total_num;
}
