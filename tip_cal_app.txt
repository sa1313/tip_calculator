<script>
function calculate() {
  var bill_num = parseFloat(document.getElementById("bill_amount").value);
  var tip_num = parseFloat(document.getElementById("tip_percent").value);
  var tip_total = (tip_num/100)*bill_num;
  var total_num = bill_num + tip_total;
  
  
  document.getElementById("tip_total").innerHTML = "$" + tip_total;
  document.getElementById("total").innerHTML = "$" + total_num;
}
